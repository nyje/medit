file(STRINGS LINGUAS langs_unstripped REGEX "^[a-zA-Z_@]+")
set(langs)
FOREACH(var ${langs_unstripped})
	string(STRIP ${var} lang)
	list(APPEND langs ${lang})
ENDFOREACH()
FOREACH(lang ${langs})
    CONFIGURE_FILE(${lang}.po ${lang}.po COPYONLY)
ENDFOREACH()
GETTEXT_PROCESS_POT_FILE(${GETTEXT_PACKAGE_GSV}.pot ALL INSTALL_DESTINATION "share/locale" LANGUAGES ${langs})
